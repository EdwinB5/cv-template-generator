% COMMANDS
\newcommand{\mpwidth}{\linewidth-\fboxsep-\fboxsep}

% COMMAND for title sections
% 1. Parameter: Text
% 2. Parameter: Color
\newcommand{\cvsection}[2]{
    % Margins
    \vspace{1cm}
    \hspace{1cm}
    \parbox[c][15pt]{1\linewidth}{ %Podría remplazarse por un minipage que puede contener mas cosas
        \textcolor{#2}{\textbf{\large{\uppercase{#1}}}}
        }
    \vspace{1cm}
    \hspace{1cm}
    \textcolor{#2}{ \rule{0.1\textwidth}{2pt} }
        
    
}

% COMMAND For Header
% 1. Paramerter: Title
% 2. Parameter: Subtitle
\newcommand*{\cvheader}[2]{
    \vspace{1cm}
    \hspace{1cm}
        \begin{center}
            \fcolorbox{bg_c_header}{bg_c_header}{
        \parbox[c][0.1\textwidth]{0.70\columnwidth}{
            \begin{center}
                \LARGE{\textcolor{bg_l_c_header}{\textbf{\uppercase{#1}}}}
                \break
                \small{\textcolor{sub_l_c_header}{\uppercase{#2}}}
            \end{center}
        }
    }
        \end{center}
    
}
